   IF paymentType = 'CASH' THEN
       CALL getPayment RETURNING amount

FUNCTION checkoutBook(book, guest)
 IF guest.accountActive NOT(guest.hasOverdueBooks) guest.outstandingFees = 0 THEN

    SET book.status to "Checked Out"

 END IF

END FUNCTION
